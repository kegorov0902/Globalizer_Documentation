202{
203 for (int i = 0; i < mOptionsCount; i++)
204 if (mOptions[i]->IsNameEqual(option->GetName()))

207 mOptions[mOptionsCount] = option;

209 mOptionsCount++;

212 if (mOptionsCount >= mOptionsSize)
213 {
214 BaseProperty<Owner>** bufOptions = new BaseProperty<Owner>*[mOptionsSize];
215 for (int i = 0; i < mOptionsSize; i++)
216 {
217 bufOptions[i] = mOptions[i];

220 delete[] mOptions;
221 delete[] mBaseTypeOptions;
222
223 mOptions = new BaseProperty<Owner>*[mOptionsSize * 2];
224 mBaseTypeOptions = new IBaseValueClass*[mOptionsSize * 2];
225
226 for (int i = 0; i < mOptionsSize * 2; i++)
227 {
228 mOptions[i] = 0;
229 mBaseTypeOptions[i] = 0;
230 }
231
232 for (int i = 0; i < mOptionsSize; i++)
233 {
234 mOptions[i] = bufOptions[i];
235 mBaseTypeOptions[i] = (IBaseValueClass*)mOptions[i];

316{
317 for (int i = 1; i < argc; i++)
318 {
319 std::string argument = argv[i];
320 for (int j = 0; j < mOptionsCount; j++)
321 {
322 if (mOptions[j]->IsNameEqual(argument))
323 {
324 if (!mOptions[j]->IsPreChange())

327 {
328 if (!mOptions[j]->IsFlag())
329 {
330 i++;
331 if (i < argc)

333 std::string value = argv[i];
334 mOptions[j]->FromString(value);
335 mOptions[j]->SetIsReadValue(true);
336 break;
337 }
338 }
339 else
340 {
341 mOptions[j]->FromString("1");
342 mOptions[j]->SetIsReadValue(true);

616template <class Owner>
617void BaseParameters<Owner>::CombineOptions(IBaseValueClass** otherOptions, int count)
618{
619 int newOtherOptionsCount = mOtherOptionsCount + count;
620 if (newOtherOptionsCount >= mOtherOptionsSize)
621 {
622 IBaseValueClass** bufOptions = new IBaseValueClass*[mOtherOptionsSize];
623 for (int i = 0; i < mOtherOptionsSize; i++)
624 {
625 bufOptions[i] = mOtherOptions[i];
626 }
627
628 delete[] mOtherOptions;
629
630 mOtherOptions = new IBaseValueClass*[mOtherOptionsSize * 2];
631
632 for (int i = 0; i < mOtherOptionsSize * 2; i++)
633 {
634 mOtherOptions[i] = 0;
635 }
636
637 for (int i = 0; i < mOtherOptionsSize; i++)
638 {
639 mOtherOptions[i] = bufOptions[i];
640 }
641 mOtherOptionsSize = mOtherOptionsSize * 2;
642
643 delete[] bufOptions;
644 }
645
646 for (int i = 0; i < count; i++)
647 {
648 mOtherOptions[mOtherOptionsCount] = otherOptions[i];
649
650 for (int j = 0; j < mOptionsCount; j++)
651 {
652 if (mOtherOptions[mOtherOptionsCount]->IsNameEqual(mOptions[j]->GetName()))
653 {
654 std::string oldBaseParameterValue = mOptions[j]->ToString();
655 std::string oldOtherParameterValue = mOtherOptions[mOtherOptionsCount]->ToString();
656 if (!mOptions[j]->IsPreChange() && !mOtherOptions[mOtherOptionsCount]->IsPreChange())
657 {
658 if (oldBaseParameterValue != oldOtherParameterValue)
659 {
660 mOptions[j]->FromString(oldOtherParameterValue);
661 std::string newBaseParameterValue = mOptions[j]->ToString();
662
663 if (newBaseParameterValue == oldBaseParameterValue)
664 {
665 mOtherOptions[mOtherOptionsCount]->FromString(oldBaseParameterValue);
666 }
667 }
668 }
669 break;
670 }
671 }
672
673 mOtherOptionsCount++;
674 }
