232template <class FType>
233double ConstrainedProblemGenerator<FType>::MaxFunctionCalculate(double* y)
234{
235 double x[MAX_TRIAL_DIMENSION];
236 if (GetRealNumberOfConstraints() > 0)
237 {
238 for (int j = 0; j < mDim; j++)
239 {
240 x[j] = (y[j] - (*mvShift)[0][j]) / (*mvZoomRatios)[0];
241 }
242 double res = mPConstraints[0]->Calculate(x);
243 for (int i = 1; i < GetRealNumberOfConstraints(); i++)
244 {
245 for (int j = 0; j < mDim; j++)
246 {
247 x[j] = (y[j] - (*mvShift)[i][j]) / (*mvZoomRatios)[i];
248 }
249
250 double f = mPConstraints[i]->Calculate(x);
251 if (res < f)
252 res = f;
253 }
254 return res;
255 }
256 return 0;

269template <class FType>
270double ConstrainedProblemGenerator<FType>::CalculateFunctionConstrained(const double* y, int fNumber)
271{
272 double x[MAX_TRIAL_DIMENSION];
273 if (fNumber < GetRealNumberOfConstraints())
274 {
275 for (int i = 0; i < mDim; i++)
276 {
277 x[i] = (y[i] - (*mvShift)[fNumber][i]) / (*mvZoomRatios)[fNumber];
278 }
279 return mPConstraints[fNumber]->Calculate(x) - (*mmQ)[fNumber];
280 }
281 else
282 {
283 if (mmIsImprovementOfTheObjective)
284 {
285 double resultCoefficient = 0;
286
287 for (int j = 0; j < GetRealNumberOfConstraints(); j++)
288 {
289 for (int i = 0; i < mDim; i++)
290 {
291 x[i] = (y[i] - (*mvShift)[j][i]) / (*mvZoomRatios)[j];
292 }
293 double f = mPConstraints[j]->Calculate(x) - (*mmQ)[j];
294 double fVal = std::max(f, 0.0);
295 resultCoefficient += (*improvementCoefficients)[j] * (fVal * fVal * fVal);
296 }
297 double result = 0;
298 result = mPConstraints[fNumber]->Calculate(y) - resultCoefficient;
299 return result;
300 }
301 else
302 {
303 double boundaryShift = 0; // âåëè÷èíà ñäâèãà
304 double* objectiveMin = new double[mDim];
305 for (int j = 0; j < mDim; j++)
306 {
307 objectiveMin[j] = 0;
308 x[j] = y[j];
309 }
310 GetOptimumPoint(objectiveMin);
311
312 int coordinateNum = 0; // êîîðäèíàòà, ïî êîòîðîé ïðîèñõîäèò ñäâèã
313 for (int i = 0; i < mDim; i++)
314 {
315 if (mvBoundaryShift[i] != 0)
316 {
317 coordinateNum = i;
318 boundaryShift = mvBoundaryShift[i];
319 }
320 }
321 // ïðåîáðàçîâàíèå êîîðäèíàò òàêèì îáðàçîì, ÷òîáû òî÷êà îïòèìóìà îêàçàëàñü íà ãðàíèöå äîïóñòèìîé îáëàñòè
322 if (boundaryShift > 0)
323 {
324 if ((y[coordinateNum] >= objectiveMin[coordinateNum]) &&
325 (y[coordinateNum] < objectiveMin[coordinateNum] + boundaryShift))
326 {
327 x[coordinateNum] = objectiveMin[coordinateNum] + boundaryShift -
328 (objectiveMin[coordinateNum] + boundaryShift - y[coordinateNum]) * 2;
329 }
330 else if ((y[coordinateNum] > objectiveMin[coordinateNum] - 2 * boundaryShift) &&
331 (y[coordinateNum] < objectiveMin[coordinateNum]))
332 {
333 x[coordinateNum] = objectiveMin[coordinateNum] - 2 * boundaryShift +
334 (y[coordinateNum] - (objectiveMin[coordinateNum] - 2 * boundaryShift)) / 2;
335 }
336 }
337 else if (boundaryShift < 0)
338 {
339 if ((y[coordinateNum] > objectiveMin[coordinateNum]) &&
340 (y[coordinateNum] < objectiveMin[coordinateNum] - 2 * boundaryShift))
341 {
342 x[coordinateNum] = objectiveMin[coordinateNum] - 2 * boundaryShift -
343 (objectiveMin[coordinateNum] - 2 * boundaryShift - y[coordinateNum]) / 2;
344 }
345 else if ((y[coordinateNum] > objectiveMin[coordinateNum] + boundaryShift) &&
346 (y[coordinateNum] < objectiveMin[coordinateNum]))
347 {
348 x[coordinateNum] = objectiveMin[coordinateNum] + boundaryShift +
349 (y[coordinateNum] - (objectiveMin[coordinateNum] + boundaryShift)) * 2;
350 }
351 }
352 delete[] objectiveMin;
353
354 return mPConstraints[fNumber]->Calculate(x);
355 }
356 }

359template <class FType>
360double ConstrainedProblemGenerator<FType>::CalculateRHS(double delta, int m, double Epsilon, int maxM)
361{
362 double rhs = 0;
363 double* objectiveMin = new double[mDim];
364 unsigned dimension = mDim;
365 double hmin = 0;
366
367 for (int j = 0; j < mDim; j++)
368 objectiveMin[j] = 0;
369 GetOptimumPoint(objectiveMin);
370
371 hmin = mPObjective->GetOptimumValue();
372 //mPObjective->GetOptimumValue(hmin);
373
374 double hmax = hmin;
375 double d = 0;
376 //ìíîãîìåðíàÿ ðåøåòêà, â óçëàõ - èñïûòàíèÿ
377 int* size = new int[dimension];//êîë-âî êîëâî òî÷åê íà ðàçìåðíîñòü
378 double* step = new double[dimension];//øàã ïî êàæäîé ðàçìåðíîñòè
379 int sumn = 1;//÷èñëî èñïûòàíèé
380
381 double* a = new double[dimension];
382 double* b = new double[dimension];
383 mPObjective->GetBounds(a, b);
384 double multiplyingLength = 1;
385 for (unsigned i = 0; i < dimension; i++)
386 {
387 d = (b[i] - a[i]);
388 size[i] = (int)ceil(d / Epsilon) + 1;
389 step[i] = d / (size[i] - 1);
390 multiplyingLength = multiplyingLength * d;
391 sumn *= (size[i]);
392 }
393
394 if ((sumn > maxM) || (sumn <= 0))
395 {
396 multiplyingLength = multiplyingLength / maxM;
397 Epsilon = pow(multiplyingLength, 1.0 / (double)dimension);
398 sumn = 1;
399 multiplyingLength = 1;
400
401 for (unsigned i = 0; i < dimension; i++)
402 {
403 d = (b[i] - a[i]);
404 size[i] = (int)ceil(d / Epsilon) + 1;
405 step[i] = d / (size[i] - 1);
406 sumn *= (size[i]);
407 }
408
409 }
410
411 double* f = new double[sumn];//çíà÷åíèå ôóíêöèè
412 double* yArray = new double[dimension*omp_get_max_threads()];
413
414#pragma omp parallel for num_threads(omp_get_max_threads())
415 for (int i = 0; i < sumn; i++)
416 {
417 double w;
418 int z = i;
419 double* y = yArray + omp_get_thread_num()*dimension;
420 //Âû÷èñëÿåì êîîðäèíàòû òî÷êè èñïûòàíèÿ
421 for (unsigned j = 0; j < dimension; j++)
422 {
423 w = z % size[j];//îïðåäåëÿåì íîìåð óçëà íà i-îé ðàçìåðíîñòè
424 y[j] = a[j] + w * step[j];//ëåâàÿ ãðàíèöà + íîìåð óçëà íà i-îé ðàçìåðíîñòè * øàã íà i-îé ðàçìåðíîñòè
425 z = z / size[j];//äëÿ âû÷èñëåíèÿ íîìåðà óçëà íà ñëåäóþùåé ðàçìåðíîñòè
426 }
427 //ïðîâîäèì èñïûòàíèå
428 f[i] = FunctionCalculate(y);
429 hmax = std::max(f[i], hmax);
430 hmin = std::min(f[i], hmin);
431 }
432
433 double* h1 = new double[m];
434 double* h2 = new double[m];
435 int* p = new int[m];
436 int* s = new int[m];
437
438 double deltah = (hmax - hmin) / m;
439
440 for (int i = 0; i < m; i++)
441 {
442 h1[i] = hmin + i * deltah;
443 h2[i] = hmin + (i + 1) * deltah;
444 p[i] = 0;
445 s[i] = 0;
446 }
447
448 for (int i = 0; i < sumn; i++)
449 for (int j = 0; j < m; j++)
450 if ((f[i] >= h1[j]) && (f[i] <= h2[j]))
451 {
452 p[j] ++;
453 break;
454 }
455
456 s[0] = p[0];
457 for (int i = 1; i < m; i++)
458 {
459 s[i] = s[i - 1] + p[i];
460 }
461
462 double smax = s[m - 1];
463 double g = delta * smax;
464 for (int i = 0; i < m; i++)
465 {
466 if (s[i] >= g)
467 {
468 rhs = h2[i];
469 break;
470 }
471 }
472
473 double dm = delta;
474 if (dm == 0)
475 dm += 0.1;
476 dm = dm * (hmax - hmin);
477
478 double criticalValue = FunctionCalculate(objectiveMin);
479
480 if (rhs < criticalValue)
481 {
482 std::cout << "Q was changed from " << rhs << " to " << criticalValue + dm << "\n";
483 rhs = criticalValue + dm;
484 }
485
486 delete[] size;
487 delete[] step;
488 delete[] a;
489 delete[] b;
490 delete[] f;
491 delete[] yArray;
492
493 delete[] h1;
494 delete[] h2;
495 delete[] p;
496 delete[] s;
497
498 return rhs;
499}

514template <class FType>
515ConstrainedProblem<FType> ConstrainedProblemGenerator<FType>::
516GenerateProblem(int generateOptions, int boundarySearchPrecision)
517{
518 mIsShift = bool(generateOptions & SHIFT);
519 mIsZoom = bool(generateOptions & ZOOM);
520
521 mIsBoundaryShift = bool(generateOptions & BOUNDARY_SHIFT);
522 mBoundarySearchPrecision = boundarySearchPrecision;
523
525 FType** mPFunction = new FType*[mNumberOfConstraints + 1];
526 for (int i = 0; i < mNumberOfConstraints; i++)
527 {
528 mPFunction[i] = mPConstraints[i];
529 }
530 mPFunction[mNumberOfConstraints] = mPObjective;
531
532 double* objectiveMin = new double[mDim];
533
535 mmQ = new double*[1];
536 (*mmQ) = new double[mNumberOfConstraints + 1];
537
539 double* mZoomRatios = new double[mNumberOfConstraints + 1];
541 double** mShift = new double*[mNumberOfConstraints + 1];
542
543 for (int i = 0; i < GetRealNumberOfFunctions(); i++)
544 {
545 mZoomRatios[i] = 1.0;
546
547 mShift[i] = new double[mDim];
548 for (int j = 0; j < mDim; j++)
549 {
550 mShift[i][j] = 0.0;
551 }
552 }
553
555 double* mBoundaryShift = new double[mDim];
556 for (int i = 0; i < mDim; i++)
557 {
558 mBoundaryShift[i] = 0.0;
559 }
560 mvBoundaryShift = mBoundaryShift;
561 mvZoomRatios = &mZoomRatios;
562 mvShift = &mShift;
563
564 mmIsImprovementOfTheObjective = bool(generateOptions & IMPROVE_OBJECTIVE);
565 mIsTotalDelta = bool(generateOptions & TOTAL_DELTA);
567 improvementCoefficients = new double*[1];
568 *improvementCoefficients = new double[mNumberOfConstraints + 1];
569 for (int i = 0; i < mNumberOfConstraints; i++)
570 (*improvementCoefficients)[i] = mImprovementOfTheObjective[i];
571 (*improvementCoefficients)[mNumberOfConstraints] = 0;
572
573 for (int i = 0; i < GetRealNumberOfFunctions(); i++)
574 {
575 InitFunc(mPFunction[i], i);
576 }
577
578 if (mIsZoom)
579 {
580 SetZoom();
581 }
582
583 if (mIsShift)
584 {
585 SetShift();
586 }
587
588 currentFunctionNumber = 0;
589
590 if (mIsTotalDelta)
591 {
592 double q = CalculateRHS(mConstraintsParams[0]);
593 for (int i = 0; i < GetRealNumberOfConstraints(); i++)
594 {
595 (*mmQ)[i] = q;
596 }
597 }
598 else
599 {
600 for (int i = 0; i < GetRealNumberOfConstraints(); i++)
601 {
602 if (mNeedTuneParam[i])
603 {
604 currentFunctionNumber = i;
605 currentFunction = mPFunction[i];
606 (*mmQ)[i] = CalculateRHS(mConstraintsParams[i]);
607 }
608 else
609 {
610 (*mmQ)[i] = mConstraintsParams[i];
611 }
612 }
613 }
614
615 if (mIsBoundaryShift)
616 {
617 SetBoundaryShift();
618 }
619
620 return ConstrainedProblem<FType>(mPFunction, mmQ, mZoomRatios, mShift,
621 mmIsImprovementOfTheObjective, improvementCoefficients, mNumberOfConstraints,
622 mNumberOfCriterions, mDim);
623}

737template <class FType>
738void ConstrainedProblemGenerator<FType>::SetZoom()
739{
740
741 double* objectiveMin = new double[mDim];
742 double* lower = new double[mDim];
743 double* upper = new double[mDim];
744 double* constraintMin = new double[mDim];
745
746 for (int j = 0; j < mDim; j++)
747 {
748 objectiveMin[j] = 0;
749 lower[j] = 0;
750 upper[j] = 0;
751 }
753 GetOptimumPoint(objectiveMin);
755 mPObjective->GetBounds(lower, upper);
756
757 double maxDistanceToBoundary = 0;
758
759 for (int k = 0; k < mDim; k++)
760 {
761 if (maxDistanceToBoundary < (objectiveMin[k] - lower[k]))
762 maxDistanceToBoundary = (objectiveMin[k] - lower[k]);
763 if (maxDistanceToBoundary < (upper[k] - objectiveMin[k]))
764 maxDistanceToBoundary = (upper[k] - objectiveMin[k]);
765 }
766
767 if (fabs(maxDistanceToBoundary) < AccuracyDouble)
768 {
769 mIsZoom = false;
770 for (int j = 0; j < GetRealNumberOfFunctions(); j++)
771 {
772 (*mvZoomRatios)[j] = 1;
773 }
774
775 delete[] objectiveMin;
776 delete[] lower;
777 delete[] upper;
778 delete[] constraintMin;
779 return;
780 }
781
782 for (int i = 0; i < GetRealNumberOfConstraints(); i++)
783 {
784 double minDistanceToBoundary = upper[0] - lower[0];
785
786 for (int j = 0; j < mDim; j++)
787 {
788 constraintMin[j] = 0;
789 }
790
791 GetConstraintOptimumPoint(constraintMin, i);
792 for (int k = 0; k < mDim; k++)
793 {
794 if (minDistanceToBoundary > (constraintMin[k] - lower[k]))
795 minDistanceToBoundary = (constraintMin[k] - lower[k]);
796 if (minDistanceToBoundary > (upper[k] - constraintMin[k]))
797 minDistanceToBoundary = (upper[k] - constraintMin[k]);
798 }
799
800 if (fabs(minDistanceToBoundary) < AccuracyDouble)
801 {
802 mIsZoom = false;
803 for (int j = 0; j < GetRealNumberOfFunctions(); j++)
804 {
805 (*mvZoomRatios)[j] = 1;
806 }
807 delete[] objectiveMin;
808 delete[] lower;
809 delete[] upper;
810 delete[] constraintMin;
811 return;
812 }
813 else
814 {
815 (*mvZoomRatios)[i] = maxDistanceToBoundary / minDistanceToBoundary;
816 }
817 }
818
819 delete[] objectiveMin;
820 delete[] lower;
821 delete[] upper;

964 }

966 }

969
970 delete[] objectiveMin;
971 delete[] tempPoint;
