204template <class Owner, class FType>
205int ProblemWithConstraints<Owner, FType>::CheckValue(int index)
206{
207 BaseProblem<Owner>::CheckValue(index);
208
209 if ((index < this->mOptionsCount) && (index >= 0))
210 {
211 if (this->mOptions[index] == &Deltas)
212 {
213 if (Deltas.GetSize() == 1)
214 this->mIsTotalDelta = true;
215 else
216 this->mIsTotalDelta = false;
217 }
218 }
219
220 if (this->mIsInit)
221 {
222 if ((constraint_count > 50) || (constraint_count < 0))
223 constraint_count = 0;
224
225 if (Deltas.GetSize() < constraint_count)
226 {
227 if (Deltas.GetSize() == 1)
228 {
229 double d = Deltas[0];
230 if ((d > 1) || (d < 0))
231 d = 0.5;
232 Deltas.SetSize(constraint_count);
233 for (int i = 0; i < constraint_count; i++)
234 Deltas[i] = d;
235 }
236 else
237 {
238 int oldSize = Deltas.GetSize();
239 double d = Deltas[0];
240 if ((d > 1) || (d < 0))
241 d = 0.5;
242 Deltas.SetSize(constraint_count);
243 for (int i = oldSize; i < constraint_count; i++)
244 Deltas[i] = d;
245 }
246 }
247
248 if (Q.GetSize() < GetRealNumberOfFunctions())
249 {
250 if (Q.GetSize() == 1)
251 {
252 double d = Q[0];
253 if ((d > 1) || (d < 0))
254 d = 0.5;
255 Q.SetSize(GetRealNumberOfFunctions());
256 for (int i = 0; i < GetRealNumberOfConstraints(); i++)
257 Q[i] = d;
258 for (int i = GetRealNumberOfConstraints(); i < GetRealNumberOfFunctions(); i++)
259 Q[i] = 0;
260 }
261 else
262 {
263 int oldSize = Q.GetSize();
264 double d = Q[0];
265 if ((d > 1) || (d < 0))
266 d = 0.5;
267 Q.SetSize(GetRealNumberOfFunctions());
268 for (int i = oldSize; i < GetRealNumberOfConstraints(); i++)
269 Q[i] = d;
270 for (int i = std::max(GetRealNumberOfConstraints(), oldSize); i < GetRealNumberOfFunctions(); i++)
271 Q[i] = 0;
272 }
273 }
274
275
276 if (ImprovementCoefficients.GetSize() < GetRealNumberOfFunctions())
277 {
278 if (ImprovementCoefficients.GetSize() == 1)
279 {
280 double d = ImprovementCoefficients[0];
281 ImprovementCoefficients.SetSize(GetRealNumberOfFunctions());
282 for (int i = 0; i < GetRealNumberOfConstraints(); i++)
283 ImprovementCoefficients[i] = d;
284 for (int i = GetRealNumberOfConstraints(); i < GetRealNumberOfFunctions(); i++)
285 ImprovementCoefficients[i] = 0;
286 }
287 else
288 {
289 int oldSize = ImprovementCoefficients.GetSize();
290 double d = ImprovementCoefficients[0];
291 ImprovementCoefficients.SetSize(GetRealNumberOfFunctions());
292 for (int i = oldSize; i < GetRealNumberOfConstraints(); i++)
293 ImprovementCoefficients[i] = d;
294 for (int i = std::max(GetRealNumberOfConstraints(), oldSize); i < GetRealNumberOfFunctions(); i++)
295 ImprovementCoefficients[i] = 0;
296 }
297 }
298
299 }
300 return 0;
301}

400template <class Owner, class FType>
401void ProblemWithConstraints<Owner, FType>::Init(int argc, char* argv[], bool isMPIInit)
402{
403 BaseProblem<Owner>::Init(argc, argv, false);
404 this->mPFunction = new FType*[GetRealNumberOfFunctions()];
405 this->mZoomRatios = new double[GetRealNumberOfFunctions()];
406 this->mShift = new double*[GetRealNumberOfFunctions()];
407 this->mBoundaryShift = new double[this->Dimension];
408 this->mvZoomRatios = &this->mZoomRatios;
409 this->mvShift = &this->mShift;
410 this->mvBoundaryShift = this->mBoundaryShift;
411
412 for (int i = 0; i < GetRealNumberOfFunctions(); i++)
413 {
414 this->mZoomRatios[i] = 1.0;
415
416 this->mShift[i] = new double[this->Dimension];
417 for (int j = 0; j < this->Dimension; j++)
418 {
419 this->mShift[i][j] = 0.0;
420 }
421 }
422 for (int i = 0; i < this->Dimension; i++)
423 {
424 this->mBoundaryShift[i] = 0.0;
425 }
426 for (int i = 0; i < GetRealNumberOfFunctions(); i++)
427 {
428 this->mPFunction[i] = new FType();
429 this->mPConstraints.push_back(this->mPFunction[i]);
430 this->InitFunc(this->mPFunction[i], i);
431 }
432 this->mPObjective = this->mPFunction[GetRealNumberOfFunctions() - 1];
433
434 if (IsZoom)
435 {
436 this->SetZoom();
437 }
438
439 if (IsShift)
440 {
441 this->SetShift();
442 }
443
444 this->mmIsImprovementOfTheObjective = IsImprovementOfTheObjective;
445 this->mmQ = new double*[1];
446 (*this->mmQ) = new double[GetRealNumberOfConstraints() + 1];
447
448 this->improvementCoefficients = new double*[1];
449 *this->improvementCoefficients = new double[GetRealNumberOfConstraints() + 1];
450 for (int i = 0; i < GetRealNumberOfConstraints(); i++)
451 (*this->improvementCoefficients)[i] = ImprovementCoefficients[i];
452 (*this->improvementCoefficients)[GetRealNumberOfConstraints()] = 0;
453
454 this->currentFunctionNumber = 0;
455 if (Deltas.GetIsReadValue())
456 {
457 /*for (int j = 0; j < Dimension; j++)
458 objectiveMin[j] = 0;*/
459 //GetOptimumPoint(objectiveMin);
460 if (this->mIsTotalDelta)
461 {
462 double q = this->CalculateRHS(Deltas[0]);
463 for (int i = 0; i < GetRealNumberOfConstraints(); i++)
464 {
465 Q[i] = q;
466 (*this->mmQ)[i] = Q[i];
467 }
468 }
469 else
470 {
471 for (int i = 0; i < GetRealNumberOfConstraints(); i++)
472 {
473 this->currentFunctionNumber = i;
474 this->currentFunction = this->mPFunction[i];
475 Q[i] = this->CalculateRHS(Deltas[i]);
476 (*this->mmQ)[i] = Q[i];
477 }
478 }
479 }
480
481 if (IsBoundaryShift)
482 {
483 this->SetBoundaryShift();
484 }
485
486}
