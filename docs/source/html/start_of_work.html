    #include "Globalizer.h"

double MishrasBirdFunctionals(const double* y, int fNumber){
    double sin1 = sin(y[1]), cos0 = cos(y[0]);
    double result = 0.0;

    if (fNumber == 1)
        result = sin1 * exp(pow(1 - cos0, 2)) + cos0 * exp(pow(1 - sin1, 2)) + pow(y[0] - y[1], 2);
    else
        result = pow(y[0] + 5, 2) + pow(y[1] + 5, 2) - 25;
    return result;
}

int main(int argc, char* argv[]){
    GlobalizerInitialization(argc, argv);
    parameters.Dimension = 2;
    IProblem* problem = new ProblemFromFunctionPointers(
        parameters.Dimension,
        { -10.0, -6.5 },
        { 0.0, 0.0 },
        MishrasBirdFunctionals,
        2,
        true,
        -106.765,
        { -3.1302468, -1.5821422 }
    );
    problem->Initialize();
    Solver solver(problem);
    if (solver.Solve() != SYSTEM_OK)
        throw EXCEPTION("Error: solver.Solve crash!!!");
    return 0;
}

// ------------------------------------------------------------------------------------------------
double SampleProblem_15_12_2025::CalculateFunctionals(const std::vector<double>& x,
std::vector<std::string>& u, int fNumber)
{
double res = 0.0;
double x1 = x[0], x2 = x[1], x3 = x[2];
double y1 = u[0].c_str()[0] - '0', y2 = u[1].c_str()[0] - '0',
y3 = u[2].c_str()[0] - '0', y4 = u[3].c_str()[0] - '0';

switch (fNumber)
{
case 0: // constraint 1
res = y1 + y2 + y3 + x1 + x2 + x3 - 5.0;
break;
case 1: // constraint 2
res = y3 * y3 + x1 * x1 + x2 * x2 + x3 * x3 - 5.5;
break;
case 2: // constraint 3
res = y1 + x1 - 1.2;
break;
case 3: // constraint 4
res = y2 + x2 - 1.8;
break;
case 4: // constraint 5
res = y3 + x3 - 2.5;
break;
case 5: // constraint 6
res = y4 + x1 - 1.2;
break;
case 6: // constraint 7
res = y2 * y2 + x2 * x2 - 1.64;
break;
case 7: // constraint 8
res = y3 * y3 + x3 * x3 - 4.25;
break;
case 8: // constraint 9
res = y2 * y2 + x3 * x3 - 4.64;
break;
case 9: // criterion
res = (y1 - 1.0) * (y1 - 1.0) + (y2 - 2.0) * (y2 - 2.0) + (y3 - 1.0) * (y3 - 1.0) - log(y4 + 1.0) + (x1 - 1.0) * (x1 - 1.0) + (x2 - 2.0) * (x2 - 2.0) + (x3 - 3.0) * (x3 - 3.0);
break;
}
return res;
}

    1.1.2. Запуск с настройкой параметров

    Подсказку по имеющимся параметрам можно вызвать через команду:

    python start_py.py --help
python start_py.py --h

    Описание всех параметров:

    usage: start_py.py [-h] [-pfn PLOTFILENAME] [-ft {LevelLayers,Surface}] 
                   [-ct {ObjectiveFunction,Approximation,Interpolation,ByPoints,OnlyPoints}] 
                   [-ctc {ObjectiveFunction,Approximation,Interpolation,ByPoints,OnlyPoints}] 
                   [-eps EPSILON] [-x1 X1] [-x2 X2] [-lvls LEVELS] 
                   [-grido OBJECTIVEGRIDSIZE] [-gridc CONSTRAINTSGRIDSIZE] 
                   [-b BOUNDS] [-ShowFigure] [-HideTrialsPoints] 
                   [-PointsBelowGraph] [-FillFeasibleRegion] [-1D]
                   SourceFilesPath ProblemFileName TrialsFileName

    Позиционные аргументы:

    • SourceFilesPath — путь до файлов для чтения информации о задаче и точек испытаний.
    • ProblemFileName — файл с информацией о задаче.
    • TrialsFileName — файл с точками испытаний.

    Опциональные параметры:

    • -h, --help — показать справку и выйти.
    • -pfn PLOTFILENAME, --PlotFileName PLOTFILENAME — имя файла для сохранения изображения.
    • -ft {LevelLayers,Surface}, --FigureType {LevelLayers,Surface} — тип визуализации целевой функции.
    • -ct {ObjectiveFunction,Approximation,Interpolation,ByPoints,OnlyPoints}, --CalcsType {ObjectiveFunction,Approximation,Interpolation,ByPoints,OnlyPoints} — тип вычислений для визуализации целевой функции.
    • -eps EPSILON, --Epsilon EPSILON — точность решения.
    • -x1 X1, --x1 X1 — первый номер параметра для визуализации.
    • -x2 X2, --x2 X2 — второй номер параметра для визуализации.
    • -lvls LEVELS, --Levels LEVELS — количество линий уровня в режиме LevelLayers.
    • -grido OBJECTIVEGRIDSIZE, --ObjectiveGridSize OBJECTIVEGRIDSIZE — количество точек сетки для визуализации целевой функции.
    • -gridc CONSTRAINTSGRIDSIZE, --ConstraintsGridSize CONSTRAINTSGRIDSIZE — количество точек сетки для визуализации ограничений.
    • -b BOUNDS, --Bounds BOUNDS — границы области визуализации (в формате " {lower_x1}_{lower_x2} {upper_x1}_{upper_x2}" или " {lower_x1}_{upper_x1}"). Важно: обязателен пробел в начале, если lower_x1 < 0.
    • -ShowFigure, --ShowFigure — флаг, указывающий на необходимость открытия полученного рисунка в интерактивном окне.
    • -HideTrialsPoints, --HideTrialsPoints — флаг, указывающий на необходимость скрытия пробных точек.
    • -PointsBelowGraph, --PointsBelowGraph — флаг, указывающий на необходимость смещения пробных точек на рисунке ниже графика.
    • -FillFeasibleRegion, --FillFeasibleRegion — флаг, указывающий на необходимость заливки допустимой области цветом.
    • -1D, --1D — флаг, указывающий на необходимость отрисовки одномерного графика по x1.
