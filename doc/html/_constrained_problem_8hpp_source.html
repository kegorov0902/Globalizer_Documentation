196double ConstrainedProblemBase<FType>::CalculateFunction(const double* y, int fNumber)
197{
198 double x[MAX_TRIAL_DIMENSION];
199 if (fNumber < GetConstraintsNumber())
200 {
201 for (int i = 0; i < mDim; i++)
202 {
203 x[i] = (y[i] - mShift[fNumber][i]) / mZoomRatios[fNumber];
204 }
205 return mPFunction[fNumber]->Calculate(x) - (*mQ)[fNumber];
206 }
207 else
208 {
209 if (mIsImprovementOfTheObjective)
210 {
211 double resultCoefficient = 0;
212
213 for (int j = 0; j < GetRealNumberOfConstraints(); j++)
214 {
215 for (int i = 0; i < mDim; i++)
216 {
217 x[i] = (y[i] - mShift[j][i]) / mZoomRatios[j];
218 }
219 double f = mPFunction[j]->Calculate(x) - (*mQ)[j];
220 double fVal = std::max(f, 0.0);
221 resultCoefficient += (*mImprovementCoefficients)[j] * (fVal * fVal * fVal);
222 }
223 double result = 0;
224 result = mPFunction[fNumber]->Calculate(y) - resultCoefficient;
225 return result;
226 }
227 else
228 {
229 double boundaryShift = 0; // âåëè÷èíà ñäâèãà
230 double* objectiveMin = new double[mDim];
231 for (int j = 0; j < mDim; j++)
232 {
233 objectiveMin[j] = 0;
234 x[j] = y[j];
235 }
236 GetOptimumPoint(objectiveMin);
237
238 int coordinateNum = 0; // êîîðäèíàòà, ïî êîòîðîé ïðîèñõîäèò ñäâèã
239 for (int i = 0; i < mDim; i++)
240 {

242 {

244 boundaryShift = mBoundaryShift[i];
245 }
246 }

248 if (boundaryShift > 0)

251 (y[coordinateNum] < objectiveMin[coordinateNum] + boundaryShift))
252 {
253 x[coordinateNum] = objectiveMin[coordinateNum] + boundaryShift -
254 (objectiveMin[coordinateNum] + boundaryShift - y[coordinateNum]) * 2;
255 }
256 else if ((y[coordinateNum] > objectiveMin[coordinateNum] - 2 * boundaryShift) &&
257 (y[coordinateNum] < objectiveMin[coordinateNum]))
258 {
