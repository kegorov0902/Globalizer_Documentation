134template<typename T> inline
135OutputMessage& operator<<(OutputMessage& incomingHelper, const T& value)
136{
137 if (!incomingHelper.mIsInit)
138 {
139 incomingHelper.output = incomingHelper.output + toString(value);
140 }
141 else
142 {
143 if (incomingHelper.output != "")
144 {
145 incomingHelper.Print(incomingHelper.output);
146 incomingHelper.output = "";
147 }
148
149 if ((incomingHelper.mPrintPocessNumber == incomingHelper.mProcessNumber) ||
150 (!incomingHelper.mIsMPIInit))
151 {
152 incomingHelper.Print(value);
153 }
154 else if (incomingHelper.mPrintPocessNumber == -1)//нужно доделать
155 {
156 incomingHelper.Print(value);
157 }
158 }
159 return incomingHelper;
160}

163template<typename T> inline
164void OutputMessage::Print(const T& value)
165{
166 if (mIsPrintToFile[0])
167 {
168#ifdef _DEBUG
169 print_dbg_file << value;
170#endif
171 }
172 for (int i = 1; i < streamCount; i++)
173 {
174 if ((mLogFileName[i] != "") && (mIsPrintToFile[i]))
175 {
176 if (!mLogOutputStream[i]->is_open())
177 mLogOutputStream[i]->open(mLogFileName[i].c_str(), std::fstream::out);
178 *(mLogOutputStream[i]) << value;
179 }
180 }
181 if (mIsPrintToConsole)
182 print << value;
183}
