320template <class Type, class Owner>
321bool ParameterProperty<Type, Owner>::IsNameEqual(std::string name)
322{
323 if ((name.length() == GetFullLink().length()) || (name.length() == GetLink().length()) || (name.length() == GetName().length()))
324 {
325 std::string lName = name;
326 std::string LGetFullLink = GetFullLink();
327 std::string LGetLink = GetLink();
328 std::string LGetName = GetName();
329
330 for (int i = 0; i < lName.length(); i++)
331 lName[i] = std::tolower(lName[i]);
332
333 for (int i = 0; i < LGetFullLink.length(); i++)
334 LGetFullLink[i] = std::tolower(LGetFullLink[i]);
335
336 for (int i = 0; i < LGetLink.length(); i++)
337 LGetLink[i] = std::tolower(LGetLink[i]);
338
339 for (int i = 0; i < LGetName.length(); i++)
340 LGetName[i] = std::tolower(LGetName[i]);
341
342
343 if ((lName == LGetFullLink) || (lName == LGetLink) || (lName == LGetName))
344 return true;
345 else
346 return false;
347 }
348 else
349 return false;
350}
